```sql
-- Habilitar extensión vectorial (si no existe)
CREATE EXTENSION IF NOT EXISTS vector;

-- Tabla para embeddings de TCDS
CREATE TABLE doa_tcds_embeddings (
  id BIGSERIAL PRIMARY KEY,
  content TEXT NOT NULL,
  embedding vector(3072),
  chunk_id TEXT UNIQUE,
  parent_id TEXT,
  metadata JSONB DEFAULT '{}',
  created_at TIMESTAMPTZ DEFAULT timezone('utc', now()),
  updated_at TIMESTAMPTZ DEFAULT timezone('utc', now())
);

-- Índice vectorial para búsqueda por similitud
CREATE INDEX idx_doa_tcds_embedding
ON doa_tcds_embeddings
USING ivfflat (embedding vector_cosine_ops) WITH (lists=100);

-- Índice full-text para búsqueda híbrida
CREATE INDEX idx_doa_tcds_content_fts
ON doa_tcds_embeddings
USING GIN (to_tsvector('english', content));

-- Índices de metadata
CREATE INDEX idx_doa_tcds_chunk_id ON doa_tcds_embeddings(chunk_id);
CREATE INDEX idx_doa_tcds_parent_id ON doa_tcds_embeddings(parent_id);
CREATE INDEX idx_doa_tcds_created ON doa_tcds_embeddings(created_at);

-- Función de búsqueda vectorial
CREATE OR REPLACE FUNCTION match_doa_tcds(
  query_embedding vector(3072),
  match_count INT DEFAULT 10,
  filter_code TEXT DEFAULT NULL
)
RETURNS TABLE (
  id BIGINT,
  content TEXT,
  metadata JSONB,
  similarity FLOAT
)
LANGUAGE plpgsql
AS $$
BEGIN
  RETURN QUERY
  SELECT
    t.id,
    t.content,
    t.metadata,
    1 - (t.embedding <=> query_embedding) AS similarity
  FROM doa_tcds_embeddings t
  WHERE (filter_code IS NULL OR t.metadata->>'official_code' = filter_code)
  ORDER BY t.embedding <=> query_embedding
  LIMIT match_count;
END;
$$;

-- Función de búsqueda híbrida (vector + full-text)
CREATE OR REPLACE FUNCTION hybrid_search_doa_tcds(
  query_embedding vector(3072),
  query_text TEXT,
  match_count INT DEFAULT 10
)
RETURNS TABLE (
  id BIGINT,
  content TEXT,
  metadata JSONB,
  similarity FLOAT
)
LANGUAGE plpgsql
AS $$
BEGIN
  RETURN QUERY
  WITH vector_results AS (
    SELECT t.id, t.content, t.metadata,
      1 - (t.embedding <=> query_embedding) AS vec_score
    FROM doa_tcds_embeddings t
    ORDER BY t.embedding <=> query_embedding
    LIMIT match_count * 3
  ),
  text_results AS (
    SELECT t.id, t.content, t.metadata,
      ts_rank(to_tsvector('english', t.content), plainto_tsquery('english', query_text)) AS text_score
    FROM doa_tcds_embeddings t
    WHERE to_tsvector('english', t.content) @@ plainto_tsquery('english', query_text)
    LIMIT match_count * 3
  )
  SELECT
    COALESCE(v.id, t.id) AS id,
    COALESCE(v.content, t.content) AS content,
    COALESCE(v.metadata, t.metadata) AS metadata,
    (COALESCE(v.vec_score, 0) * 0.7 + COALESCE(t.text_score, 0) * 0.3) AS similarity
  FROM vector_results v
  FULL OUTER JOIN text_results t ON v.id = t.id
  ORDER BY similarity DESC
  LIMIT match_count;
END;
$$;

COMMENT ON TABLE doa_tcds_embeddings IS 'Embeddings de documentos TCDS para búsqueda semántica RAG en DOA Hub';
```

   ```typescript
   const RAG_BASE_URL = process.env.NEXT_PUBLIC_RAG_API_URL || 'http://localhost:3002'

   let authToken: string | null = null

   export async function ragLogin(username: string, password: string) {
     const form = new URLSearchParams({ username, password })
     const res = await fetch(`${RAG_BASE_URL}/api/auth/login`, {
       method: 'POST',
       body: form,
     })
     const data = await res.json()
     authToken = data.access_token
     return data
   }

   export async function ragFetch(path: string, options: RequestInit = {}) {
     const headers: Record<string, string> = {
       ...(options.headers as Record<string, string> || {}),
     }
     if (authToken) headers['Authorization'] = `Bearer ${authToken}`
     if (!(options.body instanceof FormData)) {
       headers['Content-Type'] = 'application/json'
     }
     const res = await fetch(`${RAG_BASE_URL}${path}`, { ...options, headers })
     if (!res.ok) throw new Error(`RAG API error: ${res.status}`)
     return res.json()
   }

   export async function ragUpload(path: string, file: File) {
     const form = new FormData()
     form.append('file', file)
     const headers: Record<string, string> = {}
     if (authToken) headers['Authorization'] = `Bearer ${authToken}`
     const res = await fetch(`${RAG_BASE_URL}${path}`, {
       method: 'POST',
       headers,
       body: form,
     })
     if (!res.ok) throw new Error(`RAG upload error: ${res.status}`)
     return res.json()
   }

   // Endpoints disponibles:
   // ragFetch('/api/dashboard')           → stats del sistema
   // ragFetch('/api/documents')           → lista documentos indexados

   // ragUpload('/api/ingest/analyze', file) → analizar PDF

   // ragFetch('/api/ingest/save', { method: 'POST', body: JSON.stringify(data) })
   // ragFetch('/api/chat', { method: 'POST', body: JSON.stringify({ message, session_id }) })
   ```
